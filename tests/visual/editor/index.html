'use strict';

/* Animal.js */

var Entity = require('back4app-entity').models.Entity;

module.exports = Entity.specify({
  name: 'Animal',
  attributes: {
    name: {
      type: 'String'
    },
    createdAt: {
      type: 'Datetime',
      default: Date.now
    }
  },
  methods: {
    getInfo: function () {
      return 'I am ' + this.name + ', created at ' + new Date(this.createdAt);
    }
  }
});
